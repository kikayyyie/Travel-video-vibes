    // 1. Change Background Color
    function changeColor() {
      const colors = ["#FFDDC1", "#FF9AA2", "#A2E1DB", "#B5EAD7", "#C7CEEA"];
      document.body.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
    }

    // 2. Typewriter Effect
function typeText() {
  const text = "Hello, this is typing effect!";
  let i = 0;
  const output = document.getElementById("output");
  output.textContent = "";
  const interval = setInterval(() => {
    output.textContent += text[i];
    i++;
    if (i === text.length) clearInterval(interval);
  }, 100);
}


    // 3. Shake Button
    function shake() {
      const btn = document.getElementById("shakeBtn");
      btn.classList.add("shake");
      setTimeout(() => btn.classList.remove("shake"), 400);
    }

    // 4. Drop Emoji
    function dropEmoji() {
      const emoji = document.createElement("div");
      emoji.textContent = "ðŸŒ¸";
      emoji.style.position = "absolute";
      emoji.style.left = Math.random() * window.innerWidth + "px";
      emoji.style.top = "0px";
      document.body.appendChild(emoji);

      let pos = 0;
      const fall = setInterval(() => {
        pos += 5;
        emoji.style.top = pos + "px";
        if (pos > window.innerHeight) clearInterval(fall);
      }, 50);
    }

    // 5. Flip Text (Toggle)
function flipText() {
  const text = document.getElementById("text");
  if (text.style.transform === "rotate(180deg)") {
    text.style.transform = "rotate(0deg)"; // balik sa normal
  } else {
    text.style.transform = "rotate(180deg)"; // baliktad
  }
}
